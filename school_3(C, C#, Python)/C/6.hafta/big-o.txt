Yurutme Zamani T(n):

    int i, j;
    for(i = 0; i < m: i++)                  // 1 -atama- + (m + 1) -dongu kosulu- + m -arttirma operatoru-
        for(j = 0; j < n; j++)              // m * (1 + (n + 1) + n)
            C[i, j] = A[i, j] + B[i, j]     // m * n

    T(n) = 2m + 2 + m *(2n + 2) + mn
    T(n) = 3mn + 4m + 2,
